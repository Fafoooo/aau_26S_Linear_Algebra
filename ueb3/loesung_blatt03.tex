\noindent
Zum Nachweis, dass $W$ ein Unterraum von $V$ ist, pr\"ufe ich drei Bedingungen:
\begin{enumerate}[nosep]
    \item Nullvektor $\mathbf{0} \in W$
    \item Abgeschlossen unter Addition: $\mathbf{u}, \mathbf{v} \in W \Rightarrow \mathbf{u} + \mathbf{v} \in W$
    \item Abgeschlossen unter Skalarmultiplikation: $\mathbf{u} \in W,\; k \in \mathbb{R} \Rightarrow k\mathbf{u} \in W$
\end{enumerate}
Wenn $W$ \emph{kein} Unterraum ist, reicht ein Gegenbeispiel, das eine Bedingung verletzt.

%% ============================================================
\subsection*{Aufgabe 10}
%% ============================================================

\textbf{(a)} $V = \mathbb{R}^4$, \;$W = \{(a,b,c,d) \in \mathbb{R}^4 : a + 2c = b - 3d\}$.

Die Bedingung ist eine homogene lineare Gleichung: $a - b + 2c + 3d = 0$.

\textbf{Nullvektor:}
$0 + 2 \cdot 0 = 0 - 3 \cdot 0 \;\checkmark$

\textbf{Addition:}
Seien $\mathbf{u} = (a_1,b_1,c_1,d_1), \mathbf{v} = (a_2,b_2,c_2,d_2) \in W$.
\[
(a_1{+}a_2) + 2(c_1{+}c_2)
= \underbrace{(a_1{+}2c_1)}_{=b_1-3d_1} + \underbrace{(a_2{+}2c_2)}_{=b_2-3d_2}
= (b_1{+}b_2) - 3(d_1{+}d_2). \;\checkmark
\]

\textbf{Skalarmultiplikation:}
$k(a+2c) = k(b-3d) \;\Rightarrow\; ka + 2kc = kb - 3kd. \;\checkmark$

\textbf{$W$ ist ein Unterraum.} $\square$

\bigskip

%% --- 10(d) ---
\textbf{(d)} $V = \mathbb{R}^3$, \;$W = \{(a,b,c) \in \mathbb{R}^3 : |a| \leq |c|\}$.

\textbf{Gegenbeispiel (Addition):}

$\mathbf{u} = (1, 0, -1) \in W$ \;($|1| \leq |-1|$, also $1 \leq 1$ \checkmark)

$\mathbf{v} = (1, 0, 1) \in W$ \;($|1| \leq |1|$ \checkmark)

$\mathbf{u} + \mathbf{v} = (2, 0, 0)$. \;Ist $|2| \leq |0|$? $\;2 \leq 0$? \textbf{Nein.}

Die $a$-Komponenten addieren sich zu $2$, aber die $c$-Komponenten heben sich auf zu $0$.

\textbf{$W$ ist kein Unterraum} (nicht abgeschlossen unter Addition). $\square$

\bigskip

%% --- 10(e) ---
\textbf{(e)} $V = \mathbb{R}^3$, \;$W = \{(a,b,c) \in \mathbb{R}^3 : ac \leq 0\}$.

\textbf{Gegenbeispiel (Addition):}

$\mathbf{u} = (1, 0, 0) \in W$ \;($1 \cdot 0 = 0 \leq 0$ \checkmark)

$\mathbf{v} = (0, 0, 1) \in W$ \;($0 \cdot 1 = 0 \leq 0$ \checkmark)

$\mathbf{u} + \mathbf{v} = (1, 0, 1)$. \;Ist $1 \cdot 1 = 1 \leq 0$? \textbf{Nein.}

Einzeln hat je eine Komponente den Wert $0$ (Produkt $= 0$), aber in der Summe sind beide positiv.

\textbf{$W$ ist kein Unterraum} (nicht abgeschlossen unter Addition). $\square$

\bigskip

%% --- 10(g) ---
\textbf{(g)} $V = \mathbb{R}^4$, \;$W = \{\ell \in \mathbb{R}^4 : \ell \text{ ist arithmetische Folge}\}$.

Eine arithmetische Folge hat die Form $(a,\, a{+}d,\, a{+}2d,\, a{+}3d)$.
\"Aquivalent: $\ell_1 - 2\ell_2 + \ell_3 = 0$ und $\ell_2 - 2\ell_3 + \ell_4 = 0$ (homogene lineare Gleichungen).

\textbf{Nullvektor:}
$(0,0,0,0)$ mit $a=0, d=0$. \;\checkmark

\textbf{Addition:}
$(a_1, a_1{+}d_1, \ldots) + (a_2, a_2{+}d_2, \ldots)
= (a_1{+}a_2,\; (a_1{+}a_2){+}(d_1{+}d_2),\; \ldots)$
$\Rightarrow$ arithmetisch mit Anfangswert $a_1{+}a_2$, Differenz $d_1{+}d_2$. \;\checkmark

\textbf{Skalarmultiplikation:}
$k(a, a{+}d, a{+}2d, a{+}3d) = (ka, ka{+}kd, ka{+}2kd, ka{+}3kd)$
$\Rightarrow$ arithmetisch mit Anfangswert $ka$, Differenz $kd$. \;\checkmark

\textbf{$W$ ist ein Unterraum.} $\square$

\bigskip
\hrule
\bigskip

%% ============================================================
\subsection*{Aufgabe 11}
%% ============================================================

\textbf{(b)} $V = \mathbb{R}^{n \times n}$, \;$W = \{A : A^2 = 0\}$.

\textbf{Gegenbeispiel (Addition) f\"ur $n = 2$:}
\[
A = \begin{pmatrix} 0 & 1 \\ 0 & 0 \end{pmatrix}, \qquad
B = \begin{pmatrix} 0 & 0 \\ 1 & 0 \end{pmatrix}.
\]

$A^2 = \begin{pmatrix} 0 & 0 \\ 0 & 0 \end{pmatrix} = 0$ \;\checkmark \qquad
$B^2 = \begin{pmatrix} 0 & 0 \\ 0 & 0 \end{pmatrix} = 0$ \;\checkmark

Aber:
\[
A + B = \begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix}, \qquad
(A+B)^2 = \begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix}
\begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix}
= \begin{pmatrix} 1 & 0 \\ 0 & 1 \end{pmatrix} = I \neq 0.
\]

\textbf{$W$ ist kein Unterraum} (nicht abgeschlossen unter Addition). $\square$

\bigskip

%% --- 11(e) ---
\textbf{(e)} $V = \mathbb{R}^{n \times n}$, \;$W = \{A : KAM = A^\top\}$, \;$K, M$ fest.

\textbf{Nullvektor:}
$K \cdot 0 \cdot M = 0 = 0^\top$. \;\checkmark

\textbf{Addition:}
Seien $A, B \in W$ (also $KAM = A^\top$, $KBM = B^\top$):
\[
K(A+B)M = KAM + KBM = A^\top + B^\top = (A+B)^\top. \;\checkmark
\]

\textbf{Skalarmultiplikation:}
\[
K(kA)M = k(KAM) = kA^\top = (kA)^\top. \;\checkmark
\]

\textbf{$W$ ist ein Unterraum.} $\square$

\bigskip
\hrule
\bigskip

%% ============================================================
\subsection*{Aufgabe 12}
%% ============================================================

$V = \mathbb{R}[x]$ (Polynome mit reellen Koeffizienten).

\medskip

\textbf{(a)} $W = \{p(x) \in \mathbb{R}[x] : p(1) = 0\}$.

\textbf{Nullvektor:} $0(1) = 0$. \;\checkmark

\textbf{Addition:}
$(p+q)(1) = p(1) + q(1) = 0 + 0 = 0$. \;\checkmark

\textbf{Skalarmultiplikation:}
$(kp)(1) = k \cdot p(1) = k \cdot 0 = 0$. \;\checkmark

\textbf{$W$ ist ein Unterraum.} $\square$

\bigskip

%% --- 12(c) ---
\textbf{(c)} $W = \{p(x) \in \mathbb{R}[x] : p(1) \cdot p(2) = 0\}$.

\textbf{Gegenbeispiel (Addition):}

$p(x) = x - 1$: \;$p(1) = 0$, $p(2) = 1$ $\;\Rightarrow\; 0 \cdot 1 = 0$ \;\checkmark

$q(x) = 2 - x$: \;$q(1) = 1$, $q(2) = 0$ $\;\Rightarrow\; 1 \cdot 0 = 0$ \;\checkmark

$(p+q)(x) = (x-1) + (2-x) = 1$ \;(konstantes Polynom).

$(p{+}q)(1) = 1$, \;$(p{+}q)(2) = 1$
$\;\Rightarrow\; 1 \cdot 1 = 1 \neq 0$.

Die Summe ist einfach die Konstante $1$ -- die ist weder bei $x=1$ noch bei $x=2$ null.

\textbf{$W$ ist kein Unterraum} (nicht abgeschlossen unter Addition). $\square$

\bigskip
\hrule
\bigskip

%% ============================================================
\subsection*{Aufgabe 13}
%% ============================================================

$V = \operatorname{FUNC}(\mathbb{R})$ (alle Funktionen $\mathbb{R} \to \mathbb{R}$).

\medskip

\textbf{(a)} $W = \{f : f(-x) = -f(x) \text{ f\"ur alle } x\}$ \;(ungerade Funktionen).

\textbf{Nullvektor:} $0(-x) = 0 = -0 = -0(x)$. \;\checkmark

\textbf{Addition:}
$(f+g)(-x) = f(-x) + g(-x) = -f(x) + (-g(x)) = -(f+g)(x)$. \;\checkmark

\textbf{Skalarmultiplikation:}
$(kf)(-x) = k \cdot f(-x) = k \cdot (-f(x)) = -(kf)(x)$. \;\checkmark

\textbf{$W$ ist ein Unterraum.} $\square$

\bigskip

%% --- 13(b) ---
\textbf{(b)} $W = \{f : f(x) \neq 0 \text{ f\"ur alle } x \neq 0\}$.

\textbf{Nullvektor:}
Die Nullfunktion $f(x) = 0$ hat $f(1) = 0$, obwohl $1 \neq 0$.
Also $f \notin W$.

\textit{Zus\"atzlich (Addition):}
$f(x) = x \in W$ und $g(x) = -x \in W$
(beide $\neq 0$ f\"ur $x \neq 0$).
Aber $(f+g)(x) = 0$ f\"ur alle $x$, also $f+g \notin W$.

\textbf{$W$ ist kein Unterraum} (Nullvektor nicht enthalten). $\square$
