\subsection*{Aufgabe 14}
%% ============================================================

\textbf{Angabe.} $A, B, C, M \in \mathbb{R}^{2 \times 2}$ mit
\[
A = \begin{pmatrix} 0 & 1 \\ 1 & 2 \end{pmatrix},\quad
B = \begin{pmatrix} 3 & 2 \\ -1 & 0 \end{pmatrix},\quad
C = \begin{pmatrix} 3 & 6 \\ 3 & 8 \end{pmatrix},\quad
M = \begin{pmatrix} 6 & 7 \\ 1 & 6 \end{pmatrix}.
\]
Ist $M \in \text{span}(A,B,C)$? Falls ja: eindeutig oder auf unendlich viele Arten?

\medskip

\textbf{Schritt 1: Gleichungssystem aus $aA + bB + cC = M$ aufstellen.}

Eintragsweiser Vergleich liefert:
\[
\begin{cases}
3b + 3c = 6 & \text{(oben-links)} \\
a + 2b + 6c = 7 & \text{(oben-rechts)} \\
a - b + 3c = 1 & \text{(unten-links)} \\
2a + 8c = 6 & \text{(unten-rechts)}
\end{cases}
\]

\textbf{Schritt 2: Erweiterte Koeffizientenmatrix und Gau\ss-Elimination.}

\[
\left(\begin{array}{ccc|c}
0 & 3 & 3 & 6 \\
1 & 2 & 6 & 7 \\
1 & -1 & 3 & 1 \\
2 & 0 & 8 & 6
\end{array}\right)
\xrightarrow{Z_1 \leftrightarrow Z_2}
\left(\begin{array}{ccc|c}
1 & 2 & 6 & 7 \\
0 & 3 & 3 & 6 \\
1 & -1 & 3 & 1 \\
2 & 0 & 8 & 6
\end{array}\right)
\xrightarrow[Z_4 - 2Z_1]{Z_3 - Z_1}
\left(\begin{array}{ccc|c}
1 & 2 & 6 & 7 \\
0 & 3 & 3 & 6 \\
0 & -3 & -3 & -6 \\
0 & -4 & -4 & -8
\end{array}\right)
\]

\[
\xrightarrow[3Z_4 + 4Z_2]{Z_3 + Z_2}
\left(\begin{array}{ccc|c}
1 & 2 & 6 & 7 \\
0 & 3 & 3 & 6 \\
0 & 0 & 0 & 0 \\
0 & 0 & 0 & 0
\end{array}\right)
\]

2 Pivots, keine Widerspruchszeile $\Rightarrow$ konsistent mit einer freien Variable.

\medskip

\textbf{Schritt 3: Allgemeine L\"osung.}

Freie Variable $c = t$.
\begin{align*}
3b + 3t = 6 &\;\Rightarrow\; b = 2 - t \\
a + 2(2-t) + 6t = 7 &\;\Rightarrow\; a = 3 - 4t
\end{align*}

\[
\fbox{\parbox{0.93\textwidth}{\centering
$M \in \text{span}(A,B,C)$ auf \textbf{unendlich viele Arten}:\\[2pt]
$(a,b,c) = (3-4t,\; 2-t,\; t),\quad t \in \mathbb{R}.$
}}
\]

\textbf{Schritt 4: Drei konkrete Linearkombinationen.}

\begin{itemize}[leftmargin=*]
\item $t = 0 \Rightarrow (a,b,c) = (3, 2, 0)$:
\[
3A + 2B = \begin{pmatrix} 0 & 3 \\ 3 & 6 \end{pmatrix} + \begin{pmatrix} 6 & 4 \\ -2 & 0 \end{pmatrix}
= \begin{pmatrix} 6 & 7 \\ 1 & 6 \end{pmatrix} = M \;\checkmark
\]

\item $t = 1 \Rightarrow (a,b,c) = (-1, 1, 1)$:
\[
-A + B + C = \begin{pmatrix} 0 & -1 \\ -1 & -2 \end{pmatrix} + \begin{pmatrix} 3 & 2 \\ -1 & 0 \end{pmatrix} + \begin{pmatrix} 3 & 6 \\ 3 & 8 \end{pmatrix}
= \begin{pmatrix} 6 & 7 \\ 1 & 6 \end{pmatrix} = M \;\checkmark
\]

\item $t = -1 \Rightarrow (a,b,c) = (7, 3, -1)$:
\[
7A + 3B - C = \begin{pmatrix} 0 & 7 \\ 7 & 14 \end{pmatrix} + \begin{pmatrix} 9 & 6 \\ -3 & 0 \end{pmatrix} - \begin{pmatrix} 3 & 6 \\ 3 & 8 \end{pmatrix}
= \begin{pmatrix} 6 & 7 \\ 1 & 6 \end{pmatrix} = M \;\checkmark
\]
\end{itemize}

\subsection*{Aufgabe 15(a) -- Polynome in $\mathbb{R}_2[x]$}
%% ============================================================

\textbf{Angabe.} Ist die Menge
\[
\{(x+1)(x-2),\; (x+1)(x-3),\; (x+1)(x-4)\}
\]
in $\mathbb{R}_2[x]$ linear unabh\"angig?

\medskip

\textbf{Schritt 1: Polynome ausmultiplizieren.}
\begin{align*}
p_1 &= (x+1)(x-2) = x^2 - x - 2 \\
p_2 &= (x+1)(x-3) = x^2 - 2x - 3 \\
p_3 &= (x+1)(x-4) = x^2 - 3x - 4
\end{align*}

\textbf{Schritt 2: Ansatz $\alpha p_1 + \beta p_2 + \gamma p_3 = 0$ und Koeffizientenvergleich.}

In der Basis $\{1, x, x^2\}$:
\[
\begin{cases}
\alpha + \beta + \gamma = 0 & (x^2) \\
-\alpha - 2\beta - 3\gamma = 0 & (x) \\
-2\alpha - 3\beta - 4\gamma = 0 & (1)
\end{cases}
\]

\textbf{Schritt 3: Gau\ss-Elimination.}

\[
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
-1 & -2 & -3 & 0 \\
-2 & -3 & -4 & 0
\end{array}\right)
\xrightarrow[Z_3 + 2Z_1]{Z_2 + Z_1}
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
0 & -1 & -2 & 0 \\
0 & -1 & -2 & 0
\end{array}\right)
\xrightarrow{Z_3 - Z_2}
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
0 & -1 & -2 & 0 \\
0 & 0 & 0 & 0
\end{array}\right)
\]

2 Pivots bei 3 Unbekannten $\Rightarrow$ nichttriviale L\"osung existiert.

\medskip

\textbf{Schritt 4: Nichttriviale Linearkombination bestimmen.}

Freie Variable $\gamma = 1$:
\begin{align*}
-\beta - 2\gamma = 0 &\;\Rightarrow\; \beta = -2 \\
\alpha + \beta + \gamma = 0 &\;\Rightarrow\; \alpha = 1
\end{align*}

\[
\fbox{\parbox{0.93\textwidth}{\centering
Menge ist \textbf{linear abh\"angig}:\\[2pt]
$1 \cdot p_1 - 2 \cdot p_2 + 1 \cdot p_3 = 0.$
}}
\]

\textbf{Probe:}
\[
(x^2 - x - 2) - 2(x^2 - 2x - 3) + (x^2 - 3x - 4)
= (1-2+1)x^2 + (-1+4-3)x + (-2+6-4) = 0 \;\checkmark
\]

\subsection*{Aufgabe 15(b) -- Funktionen $\{1, 2^x, 3^x\}$}
%% ============================================================

\textbf{Angabe.} Ist die Menge $\{1,\; 2^x,\; 3^x\}$ in $\text{FUNC}(\mathbb{R})$ linear unabh\"angig?

\medskip

\textbf{Schritt 1: Ansatz und Auswertung an drei Punkten.}

$\alpha \cdot 1 + \beta \cdot 2^x + \gamma \cdot 3^x = 0$ soll f\"ur \textbf{alle} $x \in \mathbb{R}$ gelten, also insbesondere f\"ur $x = 0, 1, 2$:

\begin{align*}
x = 0: &\quad \alpha + \beta + \gamma = 0 \\
x = 1: &\quad \alpha + 2\beta + 3\gamma = 0 \\
x = 2: &\quad \alpha + 4\beta + 9\gamma = 0
\end{align*}

\textbf{Schritt 2: Gau\ss-Elimination.}

\[
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
1 & 2 & 3 & 0 \\
1 & 4 & 9 & 0
\end{array}\right)
\xrightarrow[Z_3 - Z_1]{Z_2 - Z_1}
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
0 & 1 & 2 & 0 \\
0 & 3 & 8 & 0
\end{array}\right)
\xrightarrow{Z_3 - 3Z_2}
\left(\begin{array}{ccc|c}
1 & 1 & 1 & 0 \\
0 & 1 & 2 & 0 \\
0 & 0 & 2 & 0
\end{array}\right)
\]

3 Pivots bei 3 Unbekannten $\Rightarrow$ nur die triviale L\"osung $\alpha = \beta = \gamma = 0$.

\medskip

\textbf{Schritt 3: Folgerung.}

Wenn schon f\"ur drei konkrete Punkte nur die triviale L\"osung m\"oglich ist, dann auch f\"ur alle $x \in \mathbb{R}$ gleichzeitig.

\[
\fbox{\parbox{0.93\textwidth}{\centering
Menge ist \textbf{linear unabh\"angig}.
}}
\]

\subsection*{Aufgabe 16}
%% ============================================================

\textbf{Angabe.} F\"ur welche $k \in \mathbb{R}$ ist die Menge
\[
\{v_1, v_2, v_3\} = \{(1,4,3,2),\; (1,2,3,4),\; (2, k+2, k-2, 2)\}
\]
in $\mathbb{R}^4$ linear unabh\"angig? Beide Methoden aus der Vorlesung zeigen.

\bigskip

\subsubsection*{Methode 1: Homogene Gleichung $\alpha v_1 + \beta v_2 + \gamma v_3 = 0$}

\textbf{Schritt 1: Eintragsweise Bedingungen aufstellen.}

\[
\begin{cases}
\alpha + \beta + 2\gamma = 0 & (\text{Komp.\ 1}) \\
4\alpha + 2\beta + (k+2)\gamma = 0 & (\text{Komp.\ 2}) \\
3\alpha + 3\beta + (k-2)\gamma = 0 & (\text{Komp.\ 3}) \\
2\alpha + 4\beta + 2\gamma = 0 & (\text{Komp.\ 4})
\end{cases}
\]

\textbf{Schritt 2: Gau\ss-Elimination mit Parameter $k$.}

\[
\left(\begin{array}{ccc|c}
1 & 1 & 2 & 0 \\
4 & 2 & k+2 & 0 \\
3 & 3 & k-2 & 0 \\
2 & 4 & 2 & 0
\end{array}\right)
\xrightarrow[\substack{Z_3 - 3Z_1 \\ Z_4 - 2Z_1}]{Z_2 - 4Z_1}
\left(\begin{array}{ccc|c}
1 & 1 & 2 & 0 \\
0 & -2 & k-6 & 0 \\
0 & 0 & k-8 & 0 \\
0 & 2 & -2 & 0
\end{array}\right)
\]

\[
\xrightarrow{Z_4 + Z_2}
\left(\begin{array}{ccc|c}
1 & 1 & 2 & 0 \\
0 & -2 & k-6 & 0 \\
0 & 0 & k-8 & 0 \\
0 & 0 & k-8 & 0
\end{array}\right)
\xrightarrow{Z_4 - Z_3}
\left(\begin{array}{ccc|c}
1 & 1 & 2 & 0 \\
0 & -2 & k-6 & 0 \\
0 & 0 & k-8 & 0 \\
0 & 0 & 0 & 0
\end{array}\right)
\]

\textbf{Schritt 3: Fallunterscheidung.}

F\"ur lineare Unabh\"angigkeit m\"ussen alle drei Pivots $\neq 0$ sein. Die ersten beiden ($1$ und $-2$) sind es immer, das dritte ist $k-8$.

\[
\text{Linear unabh\"angig} \iff k - 8 \neq 0 \iff k \neq 8.
\]

\textbf{Schritt 4: Nichttriviale Kombination bei $k = 8$.}

Freie Variable $\gamma = 1$:
\begin{align*}
-2\beta + (k-6)\gamma = -2\beta + 2 = 0 &\;\Rightarrow\; \beta = 1 \\
\alpha + \beta + 2\gamma = 0 &\;\Rightarrow\; \alpha = -3
\end{align*}
Probe: $-3(1,4,3,2) + (1,2,3,4) + (2,10,6,2) = (0,0,0,0) \;\checkmark$

\bigskip

\subsubsection*{Methode 2: Vektoren als Zeilen, Rang per Gau\ss{}}

\textbf{Schritt 1: Vektoren als Zeilen anordnen.}

\[
\left(\begin{array}{cccc}
1 & 4 & 3 & 2 \\
1 & 2 & 3 & 4 \\
2 & k+2 & k-2 & 2
\end{array}\right)
\]

\textbf{Schritt 2: Gau\ss-Elimination.}

\[
\xrightarrow[Z_3 - 2Z_1]{Z_2 - Z_1}
\left(\begin{array}{cccc}
1 & 4 & 3 & 2 \\
0 & -2 & 0 & 2 \\
0 & k-6 & k-8 & -2
\end{array}\right)
\xrightarrow{2Z_3 + (k-6)Z_2}
\left(\begin{array}{cccc}
1 & 4 & 3 & 2 \\
0 & -2 & 0 & 2 \\
0 & 0 & 2(k-8) & 2(k-8)
\end{array}\right)
\]

Rechnung zu Zeile 3 (Schritt 2):
\begin{itemize}[nosep, leftmargin=*]
\item Spalte 2: $2(k-6) + (k-6) \cdot (-2) = 0$
\item Spalte 3: $2(k-8) + (k-6) \cdot 0 = 2(k-8)$
\item Spalte 4: $2 \cdot (-2) + (k-6) \cdot 2 = -4 + 2k - 12 = 2(k-8)$
\end{itemize}

\textbf{Schritt 3: Auswertung.}

Linear unabh\"angig $\iff$ keine Nullzeile $\iff 2(k-8) \neq 0 \iff k \neq 8$.

\bigskip

\[
\fbox{\parbox{0.93\textwidth}{\centering
Beide Methoden: $\{v_1, v_2, v_3\}$ ist \textbf{linear unabh\"angig} genau dann, wenn $k \neq 8$.\\[2pt]
Bei $k = 8$: $-3 v_1 + v_2 + v_3 = 0$ zeigt die Abh\"angigkeit.
}}
\]

\subsection*{Aufgabe 17(a)}
%% ============================================================

\textbf{Angabe.} Sei $\{v_1, v_2, v_3\}$ linear unabh\"angig. Ist dann auch $\{v_1+v_2,\; v_2+v_3,\; v_2+2v_3\}$ linear unabh\"angig?

\medskip

\textbf{Schritt 1: Ansatz.}

$\alpha(v_1+v_2) + \beta(v_2+v_3) + \gamma(v_2+2v_3) = 0$.

\textbf{Schritt 2: Nach $v_1, v_2, v_3$ umsortieren.}

\[
\alpha v_1 + (\alpha + \beta + \gamma) v_2 + (\beta + 2\gamma) v_3 = 0
\]

\textbf{Schritt 3: Lineare Unabh\"angigkeit von $\{v_1, v_2, v_3\}$ ausnutzen.}

Da $\{v_1, v_2, v_3\}$ linear unabh\"angig ist, m\"ussen alle drei Koeffizienten null sein:
\[
\begin{cases}
\alpha = 0 \\
\alpha + \beta + \gamma = 0 \\
\beta + 2\gamma = 0
\end{cases}
\]

\textbf{Schritt 4: System l\"osen.}
\begin{align*}
\alpha &= 0 \\
\beta + \gamma &= 0 \;\Rightarrow\; \beta = -\gamma \\
-\gamma + 2\gamma = \gamma &= 0 \;\Rightarrow\; \beta = 0
\end{align*}

Nur triviale L\"osung.

\[
\fbox{\parbox{0.93\textwidth}{\centering
Aussage (a) ist \textbf{wahr}: $\{v_1+v_2,\; v_2+v_3,\; v_2+2v_3\}$ ist linear unabh\"angig.
}}
\]

\bigskip

\subsection*{Aufgabe 17(b)}
%% ============================================================

\textbf{Angabe.} Ist $\{7v_1 - 11v_2,\; 11v_2 - 13v_3,\; 7v_1 - 13v_3\}$ notwendig linear abh\"angig?

\medskip

\textbf{Schritt 1: Summe der ersten beiden Vektoren bilden.}

\[
(7v_1 - 11v_2) + (11v_2 - 13v_3) = 7v_1 \underbrace{-11v_2 + 11v_2}_{=0} - 13v_3 = 7v_1 - 13v_3
\]

Das ist genau der dritte Vektor.

\medskip

\textbf{Schritt 2: Nichttriviale Linearkombination der Null.}

\[
(7v_1 - 11v_2) + (11v_2 - 13v_3) - (7v_1 - 13v_3) = 0
\]

Koeffizienten $(1, 1, -1)$ -- alle ungleich null $\Rightarrow$ nichttrivial.

\medskip

\textbf{Schritt 3: Folgerung.}

Dies gilt unabh\"angig von der konkreten Wahl von $v_1, v_2, v_3$.

\[
\fbox{\parbox{0.93\textwidth}{\centering
Aussage (b) ist \textbf{wahr}: Die Menge ist f\"ur jede Wahl von $v_1, v_2, v_3$ linear abh\"angig.
}}
\]
